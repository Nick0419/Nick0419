EXPLAIN PLAN FOR
SELECT DEPARTMENTS.DEPARTMENT_ID,
       DEPARTMENT_NAME,
       FIRST_NAME,
       LAST_NAME
  FROM EMPLOYEES
       INNER JOIN DEPARTMENTS
          ON     (EMPLOYEE_ID = DEPARTMENTS.MANAGER_ID)
             AND (DEPARTMENTS.DEPARTMENT_ID = EMPLOYEES.DEPARTMENT_ID)
 WHERE DEPARTMENT_NAME = 'Finance'
